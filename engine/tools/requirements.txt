# Engine-tool dependencies — code discipline gate stack per ADR 0038.
# Install with: pip install -r engine/tools/requirements.txt
#
# These tools enforce Layer 2 of the code-discipline contract
# (ruff + mypy + pytest) per engine/operations/code-discipline.md.

ruff>=0.4.0
mypy>=1.10.0
pytest>=8.0.0

# Phase 4+ — live-graph audit against the Supabase DB per ADR 0016.
# psycopg v3 (binary build, zero-toolchain install). Activated only when
# SUPABASE_DB_URL is set in the environment; absence is a soft skip
# rather than a hard-fail, so non-seed-authoring sessions are not gated
# on DB connectivity. Imported lazily inside validate_graph() to keep
# the engine-tool import surface light when the audit isn't running.
psycopg[binary]>=3.1.0

# MemPalace shared-state probe — engine/tools/probe_palace.py per ADR 0045.
# Chromadb is the embedded vector DB MemPalace stores into; the probe opens
# the palace at boot, lists collections, and counts drawers as a health
# check. Lazy import inside probe_palace.run_probe() so non-probe runs
# (validate.py, most hooks) do not pay the import cost. Pinned explicitly
# even though mempalace re-exposes it transitively, so probe_palace's
# expected version is independent of mempalace's pin.
chromadb>=0.5.0

# Phase 4.6+ — structural-reference parser (engine/tools/parse_structural_reference.py)
# per ADR 0047. BeautifulSoup4 is the HTML parser the HTMLInputAdapter uses to
# normalize SEP- and IEP-shaped HTML into StructuralElement records. Lazy
# import inside HTMLInputAdapter.extract() — non-parser engine tool runs
# (validate.py, hooks) do not pay the import cost. Future PDF input adapter
# will add `pypdf` or `pdfplumber` at the session that lands it.
beautifulsoup4>=4.12.0

# MemPalace — the project's semantic-memory store, MCP server, and Stop /
# PreCompact hook capture. Per ADR 0050 (engine), mempalace runs from the
# project venv rather than user-scope system Python. The .mcp.json command
# path and the engine/tools/hooks/mempalace-hook-wrapper.sh PATH lookup
# both resolve to the venv's bin/. Provides binaries: `mempalace` (CLI) and
# `mempalace-mcp` (MCP server entry point). Transitive deps: chromadb
# (already pinned above), tomli, pyyaml.
mempalace>=3.3.3
